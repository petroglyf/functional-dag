120 uint16_t second_two_bytes = (m_id.bits1() << 4 * 8) >> 6 * 8;
121 uint16_t last_two_bytes = (m_id.bits1() & ((255 << 8) + 255));
122
123 uint16_t first_two_bytes = (m_id.bits2() >> 6 * 8);
